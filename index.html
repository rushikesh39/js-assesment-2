Find the maximum out of three numbers. 

        const Max_out_of_three = (A,B,C) => {
            if(A>B&&A>C){
              return A;
            }
            else if(B>A&&B>C){
              return B;
            }
            else if(C>A&&C>B){
              return C;
            }
            else if(A===B&& B===C){
              return -1;
            }
        
        };
